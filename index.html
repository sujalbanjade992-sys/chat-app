`${data.name} is typing...` : ''; }); function setTarget(id, name = "Global Room") { targetId = id; targetName = name; document.getElementById('target-display').innerText = name; document.getElementById('messages').innerHTML = ''; socket.emit('request history', id); if(window.innerWidth < 768) document.getElementById('sidebar').classList.remove('open'); } socket.on('load old msgs', (docs) => { docs.forEach(msg => addMessage(msg)); }); socket.on('chat message', (msg) => { addMessage(msg); }); function sendMessage(e) { e.preventDefault(); if (!inputField.value.trim()) return; const msgData = { senderName: myName, text: inputField.value, target: targetId, time: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }) }; socket.emit('chat message', msgData); socket.emit('typing', false); // Stop typing indicator on send inputField.value = ''; } function addMessage(msg) { const isMe = msg.senderName === myName; const row = document.createElement('div'); row.className = `msg-row ${isMe ? 'sent-row' : 'received-row'}`; const isDM = msg.target !== 'global'; row.innerHTML = `
${msg.text}
${msg.senderName} • ${msg.time}
`; document.getElementById('messages').appendChild(row); document.getElementById('messages').scrollTo(0, document.getElementById('messages').scrollHeight); } socket.on('user list', (users) => { const list = document.getElementById('user-list'); list.innerHTML = ''; for (let id in users) { const el = document.createElement('div'); el.className = 'user-item'; const statusColor = id === socket.id ? '#00ff88' : '#00ff88'; // Bright green for online el.innerHTML = `
